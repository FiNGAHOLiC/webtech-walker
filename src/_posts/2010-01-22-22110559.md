jQuery1.4はhtmlとかremoveがやたら早くなってると聞いたので試しにベンチとってみた。

html()
---------------

### ソース

{% highlight javascript %}
console.time('t');
for (var i=0; i < 10000; i++) {
    $('body').html('<div>hoge</div>');
}
console.timeEnd('t');
{% endhighlight %}

remove()
------------------

### ソース

{% highlight javascript %}
var foo = $('<div>');
for (var i=0; i < 10000; i++) {
    foo.append('<div>hoge</div>');
}

console.time('t');
foo.find('div').each(function() {
    $(this).remove();
});
console.timeEnd('t');
{% endhighlight %}
